# ============================================================
#  NETLIFY.TOML - Configuração de Segurança
#  Site: https://sitedevshub.netlify.app/
# ============================================================

[build]
  publish = "."

# --- Cabeçalhos de Segurança para todas as páginas ---

[[headers]]
  for = "/*"

  [headers.values]

    # Impede o site de ser embutido em iframe de outro domínio
    X-Frame-Options = "SAMEORIGIN"

    # Impede que o navegador adivinhe o tipo de arquivo
    X-Content-Type-Options = "nosniff"

    # Proteção XSS para navegadores antigos
    X-XSS-Protection = "1; mode=block"

    # Controla o que é enviado no Referer
    Referrer-Policy = "strict-origin-when-cross-origin"

    # Força HTTPS por 1 ano
    Strict-Transport-Security = "max-age=31536000; includeSubDomains; preload"

    # Bloqueia acesso a câmera, microfone, localização, etc.
    Permissions-Policy = "camera=(), microphone=(), geolocation=(), payment=()"

    # Content Security Policy ajustada para o seu site
    Content-Security-Policy = "default-src 'self' https://sitedevshub.netlify.app; script-src 'self' 'unsafe-inline' https://cdnjs.cloudflare.com https://fonts.googleapis.com https://api.anthropic.com; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src 'self' https://fonts.gstatic.com; img-src 'self' data: https:; connect-src 'self' https://api.anthropic.com https://discord.gg; frame-src 'none'; object-src 'none'; base-uri 'self'; form-action 'self'"

# --- Proteção extra para arquivos sensíveis ---

[[headers]]
  for = "/.env"
  [headers.values]
    X-Robots-Tag = "noindex"

[[headers]]
  for = "/security.txt"
  [headers.values]
    Content-Type = "text/plain"

# --- Cache seguro para assets estáticos ---

[[headers]]
  for = "*.js"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"

[[headers]]
  for = "*.css"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"

[[headers]]
  for = "*.png"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"

[[headers]]
  for = "*.jpg"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"

[[headers]]
  for = "*.webp"
  [headers.values]
    Cache-Control = "public, max-age=31536000, immutable"

# --- Redirecionar HTTP → HTTPS automaticamente ---

[[redirects]]
  from = "http://sitedevshub.netlify.app/*"
  to   = "https://sitedevshub.netlify.app/:splat"
  status = 301
  force  = true
